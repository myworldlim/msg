// If the loader is already loaded, just stop.
if (!self.define) {
  let registry = {};

  // Used for `eval` and `importScripts` where we can't get script URL by other means.
  // In both cases, it's safe to use a global var because those functions are synchronous.
  let nextDefineUri;

  const singleRequire = (uri, parentUri) => {
    uri = new URL(uri + ".js", parentUri).href;
    return registry[uri] || (
      
        new Promise(resolve => {
          if ("document" in self) {
            const script = document.createElement("script");
            script.src = uri;
            script.onload = resolve;
            document.head.appendChild(script);
          } else {
            nextDefineUri = uri;
            importScripts(uri);
            resolve();
          }
        })
      
      .then(() => {
        let promise = registry[uri];
        if (!promise) {
          throw new Error(`Module ${uri} didn’t register its module`);
        }
        return promise;
      })
    );
  };

  self.define = (depsNames, factory) => {
    const uri = nextDefineUri || ("document" in self ? document.currentScript.src : "") || location.href;
    if (registry[uri]) {
      // Module is already loading or loaded.
      return;
    }
    let exports = {};
    const require = depUri => singleRequire(depUri, uri);
    const specialDeps = {
      module: { uri },
      exports,
      require
    };
    registry[uri] = Promise.all(depsNames.map(
      depName => specialDeps[depName] || require(depName)
    )).then(deps => {
      factory(...deps);
      return exports;
    });
  };
}

// app/api/device/ip/route.ts

import { NextResponse } from 'next/server';

// Простое кэширование
let cachedIP: string | null = null;
let lastFetched = 0;
const CACHE_DURATION = 5 * 60 * 1000; // IP обновляется не чаще 1 раза в 5 минут, если кэш ещё не истёк.

export async function GET() {
  const now = Date.now();
  
  if (cachedIP && now - lastFetched < CACHE_DURATION) {
    return NextResponse.json({ ip: cachedIP });
  }

  try {
    const goResponse = await fetch('http://localhost:8181/api/device/ip', {
      signal: AbortSignal.timeout(5000)
    });

    if (!goResponse.ok) {
      throw new Error(`Go server responded with ${goResponse.status}`);
    }

    const data = await goResponse.json();
    const ip = data.ip;
    
    if (ip && typeof ip === 'string') {
      cachedIP = ip;
      lastFetched = now;
      return NextResponse.json({ ip });
    } else {
      throw new Error('Invalid IP address received');
    }
  } catch (error) {
    console.error('Failed to fetch IP from Go server:', error);
    
    if (cachedIP) {
      return NextResponse.json({ ip: cachedIP });
    }
    
    return NextResponse.json({ ip: null });
  }
}

// app/lib/store/deviceStore.ts

import { create } from 'zustand';
import { persist } from 'zustand/middleware';

export type DeviceType = 'mobile' | 'tablet' | 'desktop' | 'unknown';

export interface DeviceInfo {
  userAgent: string;
  ipAddress: string | null;
  browserLanguage: string;
  deviceId: string;
  deviceType: DeviceType;
  timezone: string;
  timezoneUTC: string;
  timezoneOffset: number;
  platform: string;
  localTime: string;
}

interface DeviceState {
  deviceInfo: DeviceInfo;
  isDetected: boolean;
  setIPAddress: (ip: string) => void;
  detectDevice: () => Promise<void>;
  generateDeviceId: () => string;
  updateLocalTime: () => void;
}

const generateDeviceId = (): string => {
  if (typeof window === 'undefined') return 'server-side';
  
  const signals = [
    navigator.userAgent,
    navigator.language,
    screen.width,
    screen.height,
    screen.colorDepth,
    new Date().getTimezoneOffset(),
    navigator.platform
  ].join('|');

  let hash = 0;
  for (let i = 0; i < signals.length; i++) {
    const char = signals.charCodeAt(i);
    hash = ((hash << 5) - hash) + char;
    hash = hash & hash;
  }
  
  return Math.abs(hash).toString(36);
};

const getCurrentLocalTime = (): string => {
  const now = new Date();
  return now.toLocaleTimeString('ru-RU', { 
    hour: '2-digit', 
    minute: '2-digit'
    // Убираем seconds для чистоты
  });
};

const getTimezoneUTC = (offset: number): string => {
  const sign = offset <= 0 ? '+' : '-';
  const absOffset = Math.abs(offset);
  const hours = Math.floor(absOffset / 60);
  return `UTC${sign}${hours}`;
};

const detectDeviceType = (userAgent: string): DeviceType => {
  const ua = userAgent.toLowerCase();
  
  if (/mobile|iphone|ipod|android|blackberry|opera mini|opera mobi/i.test(ua)) {
    return 'mobile';
  }
  if (/tablet|ipad|android(?!.*mobile)|kindle|silk/i.test(ua)) {
    return 'tablet';
  }
  return 'desktop';
};

// получение платформы
const getPlatform = (userAgent: string): string => {
  const ua = userAgent.toLowerCase();
  
  if (ua.includes('android')) return 'Android';
  if (ua.includes('iphone') || ua.includes('ipad')) return 'iOS';
  if (ua.includes('windows')) return 'Windows';
  if (ua.includes('mac')) return 'macOS';
  if (ua.includes('linux')) return 'Linux';
  
  return 'unknown';
};

export const useDeviceStore = create<DeviceState>()(
  persist(
    (set, get) => ({
      deviceInfo: {
        userAgent: '',
        ipAddress: null,
        browserLanguage: 'en',
        deviceId: '',
        deviceType: 'unknown',
        timezone: 'UTC',
        timezoneUTC: 'UTC+0',
        timezoneOffset: 0,
        platform: 'unknown',
        localTime: '00:00'
      },
      isDetected: false,

      generateDeviceId,

      detectDevice: async () => {
        if (typeof window === 'undefined') return;

        const userAgent = navigator.userAgent;
        const deviceType = detectDeviceType(userAgent);
        const deviceId = get().generateDeviceId();
        const platform = getPlatform(userAgent);
        const timezoneOffset = new Date().getTimezoneOffset();
        const timezoneUTC = getTimezoneUTC(timezoneOffset);

        set({
          deviceInfo: {
            userAgent,
            ipAddress: null,
            browserLanguage: navigator.language || 'en',
            deviceId,
            deviceType,
            timezone: Intl.DateTimeFormat().resolvedOptions().timeZone,
            timezoneUTC,
            timezoneOffset,
            platform,
            localTime: getCurrentLocalTime()
          },
          isDetected: true
        });

        // Запрашиваем IP
        try {
          const response = await fetch('/api/device/ip');
          const data = await response.json();
          if (data.ip) {
            get().setIPAddress(data.ip);
          }
        } catch (error) {
          console.warn('Failed to fetch IP address');
        }
      },

      setIPAddress: (ip) => set((state) => ({
        deviceInfo: { 
          ...state.deviceInfo, 
          ipAddress: ip 
        }
      })),

      updateLocalTime: () => set((state) => ({
        deviceInfo: {
          ...state.deviceInfo,
          localTime: getCurrentLocalTime()
        }
      }))
    }),
    {
      name: 'device-storage',
      partialize: (state) => ({
        deviceInfo: {
          userAgent: state.deviceInfo.userAgent,
          browserLanguage: state.deviceInfo.browserLanguage,
          deviceId: state.deviceInfo.deviceId,
          deviceType: state.deviceInfo.deviceType,
          timezone: state.deviceInfo.timezone,
          timezoneUTC: state.deviceInfo.timezoneUTC,
          timezoneOffset: state.deviceInfo.timezoneOffset,
          platform: state.deviceInfo.platform
        },
        isDetected: state.isDetected
      })
    }
  )
);
