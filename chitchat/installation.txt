# installation.txt
# Полная инструкция по установке минимального Next.js PWA-приложения (чат на React/Next.js)
# Версия: Next.js 15.5.4, TypeScript, App Router, PWA с next-pwa
# Цели: Минимальное PWA с работающим сервис-воркером (SW), кнопкой установки в адресной строке (на ПК в Chrome/Edge).
# Масштабируемость: Базовая настройка для большого количества пользователей (SSR, caching от Next.js; Zustand/React Query для состояния/запросов).
# Требования: Node.js >=18 (у тебя v22.17.1), Yarn v1.22.22.
# Время установки: ~10-15 мин.

## Шаг 1: Проверка окружения
# Убедись, что Node и Yarn установлены.
node --version  # Должен быть >=18 (у тебя v22.17.1)
yarn --version  # Должен быть >=1.22 (у тебя 1.22.22)

## Шаг 2: Создание базового проекта (если не создан)
mkdir chitchat
cd chitchat
yarn create next-app .

# Настройки при создании:
# - TypeScript: Yes
# - Linter: None (ESLint: No)
# - Tailwind CSS: No
# - src/ directory: No
# - App Router: Yes
# - Turbopack: No
# - Import alias: @/* (Yes, configure as @/*)

# Это установит Next.js 15.5.4 с шаблоном App Router.

## Шаг 3: Установка зависимостей для PWA и чата
# Основные для PWA: next-pwa (для SW и precaching)
# Для чата: zustand (состояние), @tanstack/react-query (запросы, кэшинг для масштаба), framer-motion (анимации), @use-gesture/react (жесты)
yarn add next-pwa
yarn add zustand
yarn add @tanstack/react-query
yarn add framer-motion
yarn add @use-gesture/react

## Шаг 4: Установка dev-зависимостей (типы для TypeScript)
yarn add -D @types/node
yarn add -D @types/react
yarn add -D @types/react-dom
yarn add -D @types/minimatch  # Если нужны для других пакетов

# Проверка TypeScript (без ошибок):
yarn tsc

## Шаг 5: Очистка и переустановка (если проблемы)
yarn cache clean
rm -rf node_modules yarn.lock package-lock.json
yarn install
rm -rf .next/cache tsconfig.tsbuildinfo

## Шаг 6: Настройка PWA
# 6.1: Настройка next.config.js (интеграция next-pwa)
# Создай/отредактируй next.config.js в корне проекта:
const withPWA = require('next-pwa')({
  dest: 'public',  # Директория для SW и манифеста
  register: true,  # Авторегистрация SW
  skipWaiting: true,  # Активация нового SW сразу
  disable: process.env.NODE_ENV === 'development',  # Отключить в dev (включи для теста)
  runtimeCaching: true,  # Кэшинг для масштаба (оффлайн)
});

module.exports = withPWA({
  reactStrictMode: true,
  swcMinify: true,  # Оптимизация
});

# 6.2: Создай manifest.json в public/ (для PWA-иконок и установки)
# Создай файл public/manifest.json:
{
  "name": "ChitChat PWA",
  "short_name": "ChitChat",
  "description": "Scalable chat app on Next.js",
  "start_url": "/",
  "display": "standalone",
  "background_color": "#ffffff",
  "theme_color": "#000000",
  "orientation": "portrait-primary",
  "icons": [
    {
      "src": "/icon-192x192.png",
      "sizes": "192x192",
      "type": "image/png"
    },
    {
      "src": "/icon-512x512.png",
      "sizes": "512x512",
      "type": "image/png"
    }
  ]
}

# Добавь иконки: Скачай/создай icon-192x192.png и icon-512x512.png в public/ (используй генераторы вроде favicon.io).

# 6.3: Добавь мета-теги для PWA в app/layout.tsx (или pages/_document.tsx если не App Router)
# В app/layout.tsx (в корне или pages/) добавь в <head>:
import Head from 'next/head';

export default function RootLayout({ children }) {
  return (
    <html lang="en">
      <Head>
        <link rel="manifest" href="/manifest.json" />
        <meta name="theme-color" content="#000000" />
        <meta name="apple-mobile-web-app-capable" content="yes" />
        <meta name="apple-mobile-web-app-status-bar-style" content="black" />
        <link rel="apple-touch-icon" href="/icon-192x192.png" />
      </Head>
      <body>{children}</body>
    </html>
  );
}

# 6.4: Регистрация SW (next-pwa сделает автоматически после build)

## Шаг 7: Тестирование PWA
# 7.1: Dev-режим (без PWA-установки, но с SW):
yarn dev  # Запусти на http://localhost:3000

# Для теста SW в dev: В next.config.js убери disable: process.env.NODE_ENV === 'development',
# Затем используй HTTPS: yarn add -D local-ssl-proxy
# Запусти: local-ssl-proxy --source 3001 --target 3000 --cert localhost.pem --key localhost-key.pem
# (Сгенерируй self-signed cert: openssl req -x509 -newkey rsa:2048 -keyout localhost-key.pem -out localhost.pem -days 365 -nodes)

# 7.2: Prod-тест (с кнопкой установки):
yarn build
yarn start  # На http://localhost:3000 (SW сгенерируется в public/sw.js)

# Для HTTPS и установки: Разверни на Vercel/Netlify (бесплатно, auto-HTTPS).
# Или локально: Используй ngrok: ngrok http 3000 -> https://your-ngrok-url
# Открой в Chrome/Edge: Lighthouse в DevTools -> PWA -> Проверь критерии.
# Кнопка установки появится в адресной строке (справа), если: HTTPS, manifest валиден, SW работает.
# Если уже установлено: Браузер покажет "Already installed" или иконку без кнопки.

## Шаг 8: Масштабируемость и чат-функции (минимально)
# Zustand: Создай store в lib/store.ts для глобального состояния (users, messages).

Плагин для интеграции PWA в Next.js. Обеспечивает автоматическую генерацию Service Worker и поддержку офлайн-режима. Вы уже добавили его для соответствия требованиям PWA.

Библиотека для создания анимаций и переходов в React. Используется в вашем проекте для анимации меню (например, плавное открытие/закрытие). Совместима с React 19, но проверьте документацию на GitHub для возможных нюансов.

Библиотека для обработки жестов (свайпы, перетаскивание). Используется в вашем меню для поддержки свайпов на мобильных устройствах. Совместима с React 19.

"@tanstack/react-query": "^5.90.2":
Библиотека для управления асинхронными данными и кэширования API-запросов. Отлично подходит для масштабируемости, так как уменьшает количество запросов к серверу. Вы добавили её для оптимизации API в вашем чат-приложении.
